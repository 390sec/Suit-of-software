import time
import subprocess
from selenium import webdriver
from selenium.webdriver.chrome.options import Options
import pyautogui
import pyperclip

# ----------------------------
# STEP 1 : EXTRACT TEXT WITH SELENIUM
# ----------------------------

def extract_text(url, wait_time=6):
    print("Launching browser to fetch dynamic webpage...")

    # Chrome options
    chrome_options = Options()
    chrome_options.headless = True  # run in background
    chrome_options.add_argument("--no-sandbox")
    chrome_options.add_argument("--disable-dev-shm-usage")

    driver = webdriver.Chrome(options=chrome_options)
    driver.get(url)
    time.sleep(wait_time)  # wait for JavaScript content to load

    # Get full visible page text
    body_element = driver.find_element("tag name", "body")
    page_text = body_element.text

    driver.quit()

    print(f"Extracted {len(page_text)} characters of text")
    return page_text


# ----------------------------
# STEP 2 : OPEN LIBREOFFICE
# ----------------------------

def launch_writer():
    print("Starting LibreOffice Writer...")
    subprocess.Popen(["libreoffice", "--writer"])
    print("Waiting for LibreOffice to be ready...")
    time.sleep(20)  # wait for full load

    # click inside document to focus
    pyautogui.click(500, 350)
    time.sleep(1)


# ----------------------------
# STEP 3 : SEND TEXT TO WRITER
# ----------------------------

def paste_text(text):
    if not text:
        print("No text to paste.")
        return

    print("Copying text to clipboard and pasting...")
    pyperclip.copy(text)              # copy all text to clipboard
    pyautogui.hotkey("ctrl", "v")     # paste into LibreOffice
    print("Text pasted successfully!")


# ----------------------------
# MAIN PROGRAM
# ----------------------------

def main():
    url = input("Enter webpage URL: ").strip()

    # extract dynamic text
    text_content = extract_text(url)

    # launch LibreOffice Writer
    launch_writer()

    # paste extracted text
    paste_text(text_content)


if __name__ == "__main__":
    main()



import os
import time
import requests
from bs4 import BeautifulSoup
from urllib.parse import urljoin

BASE_URL = "https://library.iitmandi.ac.in/os/ebk/publisherssearch.php?page="
QUERY = "&kwdd=Springer"

START_PAGE = 76
END_PAGE = 1326

DOWNLOAD_FOLDER = "/home/honcho/PersBund"
os.makedirs(DOWNLOAD_FOLDER, exist_ok=True)

HEADERS = {
    "User-Agent": "Mozilla/5.0 (X11; Linux x86_64)"
}


session = requests.Session()
session.headers.update(HEADERS)


def clean_filename(name):
    bad = ['\\','/','*','?','"','<','>','|',':']
    for b in bad:
        name = name.replace(b, "")
    return name.strip()


def safe_request(url, retries=3):

    for attempt in range(retries):
        try:
            r = session.get(url, timeout=30)
            if r.status_code == 200:
                return r
        except:
            pass

        time.sleep(2)

    return None


def download_file(url, title):

    filename = clean_filename(title) + ".pdf"
    path = os.path.join(DOWNLOAD_FOLDER, filename)

    if os.path.exists(path):
        print("✓ Already exists:", filename)
        return

    try:
        r = session.get(url, stream=True, timeout=60)

        if r.status_code != 200:
            print("✖ Bad response:", url)
            return

        print("⬇ Downloading:", filename)

        with open(path, "wb") as f:
            for chunk in r.iter_content(8192):
                if chunk:
                    f.write(chunk)

        print("✔ Saved:", filename)

    except Exception as e:
        print("✖ Download failed:", e)


for page in range(START_PAGE, END_PAGE + 1):

    page_url = f"{BASE_URL}{page}{QUERY}"

    print("\n===== PAGE", page, "=====")

    page_response = safe_request(page_url)

    if not page_response:
        print("✖ Could not load page")
        continue

    soup = BeautifulSoup(page_response.text, "html.parser")

    rows = soup.select("table tr")

    for row in rows:

        link = row.find("a")

        if not link:
            continue

        title = link.text.strip()

        detail_url = urljoin(page_url, link.get("href", ""))

        detail_response = safe_request(detail_url)

        if not detail_response:
            print("Detail page failed")
            continue

        detail_soup = BeautifulSoup(detail_response.text, "html.parser")

        pdf_link = None

        for a in detail_soup.find_all("a"):
            href = a.get("href", "")
            if ".pdf" in href.lower():
                pdf_link = urljoin(detail_url, href)
                break

        if pdf_link:
            download_file(pdf_link, title)
            time.sleep(1)

    time.sleep(2)


print("\n✔ All pages finished")

import os
import time
import re
import requests
from bs4 import BeautifulSoup
from urllib.parse import urljoin

BASE_URL = "https://library.iitmandi.ac.in/os/ebk/publisherssearch.php?page="
QUERY = "&kwdd=Springer"

START_PAGE = 76
END_PAGE = 1326

DOWNLOAD_FOLDER = "/home/honcho/PersBund"
os.makedirs(DOWNLOAD_FOLDER, exist_ok=True)

HEADERS = {
    "User-Agent": "Mozilla/5.0 (X11; Linux x86_64)"
}

session = requests.Session()
session.headers.update(HEADERS)


def clean_filename(name):
    bad = ['\\','/','*','?','"','<','>','|',':']
    for b in bad:
        name = name.replace(b, "")
    return name.strip()


def safe_request(url, retries=3):

    for _ in range(retries):
        try:
            r = session.get(url, timeout=30)
            if r.status_code == 200:
                return r
        except:
            pass

        time.sleep(2)

    return None


def extract_biblio(row_text):
    """Extract Biblio number"""
    match = re.search(r'\b\d{5}\b', row_text)
    if match:
        return int(match.group())
    return None


def download_file(url, title, biblio):

    filename = f"{biblio}_{clean_filename(title)}.pdf"
    path = os.path.join(DOWNLOAD_FOLDER, filename)

    if os.path.exists(path):
        print("✓ Already exists:", filename)
        return

    try:
        r = session.get(url, stream=True, timeout=60)

        if r.status_code != 200:
            print("✖ Bad response:", url)
            return

        print(f"⬇ Downloading [{biblio}] :", filename)

        with open(path, "wb") as f:
            for chunk in r.iter_content(8192):
                if chunk:
                    f.write(chunk)

        print("✔ Saved:", filename)

    except Exception as e:
        print("✖ Download failed:", e)


for page in range(START_PAGE, END_PAGE + 1):

    page_url = f"{BASE_URL}{page}{QUERY}"

    print(f"\n========== PAGE {page} ==========")

    page_response = safe_request(page_url)

    if not page_response:
        print("✖ Could not load page")
        continue

    soup = BeautifulSoup(page_response.text, "html.parser")

    rows = soup.select("table tr")

    biblio_rows = []

    # Extract Biblio numbers first
    for row in rows:

        text = row.get_text(" ", strip=True)
        biblio = extract_biblio(text)

        link = row.find("a")

        if biblio and link:
            biblio_rows.append((biblio, row, link))

    # Sort by Biblio number
    biblio_rows.sort(key=lambda x: x[0])

    for biblio, row, link in biblio_rows:

        title = link.text.strip()

        detail_url = urljoin(page_url, link.get("href", ""))

        print(f"Processing Biblio {biblio}")

        detail_response = safe_request(detail_url)

        if not detail_response:
            print("Detail page failed")
            continue

        detail_soup = BeautifulSoup(detail_response.text, "html.parser")

        pdf_link = None

        for a in detail_soup.find_all("a"):
            href = a.get("href", "")
            if ".pdf" in href.lower():
                pdf_link = urljoin(detail_url, href)
                break

        if pdf_link:
            download_file(pdf_link, title, biblio)

        time.sleep(1)

    time.sleep(2)

print("\n✔ Finished all pages sequentially.")
